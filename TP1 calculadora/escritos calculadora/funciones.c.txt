#include "funciones.h"

float sumar(float x, float y)
{
    float resultado;
    resultado=x+y;

    return resultado;
}

float restar(float x, float y)
{
    float resultado;
    resultado=x-y;

    return resultado;
}

float dividir(float x, float y)
{
    float resultado;
    resultado=x/y;

    return resultado;
}

float multiplicar(float x, float y)
{
    float resultado;
    resultado=x*y;

    return resultado;
}

float factorial(float x)
{
    float resultado;
    resultado=x*(x-1);

    return resultado;
}
